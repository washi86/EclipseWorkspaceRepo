 $\ddot x=t^{-1}f(y/t,\dot y),\;\ddot y=0$\\

\ddot \mu \ne 0$ & $\ddot x=f(t)\dot x,\;
\ddot y=-\ddot \mu (t)\ln \dot x+g(t)$\\ 
\cline{2-4}
& & $X_1=p,\;X_2=\mu (t)\E^xq,\;X_3=q,\;X_4=tq,\;\mu \ne 0$ &

\cline{3-4}

$\ddot x =\dot x^2+\left (\displaystyle{\frac{\dot \eta}{\eta}}+\eta f(t)\right )
\dot x+f(t)\dot \eta -\displaystyle{\frac{\ddot \eta}{\eta}+\frac{\dot \eta^2}
{\eta^2}}$\\
&$\LL_{4,2}^7$  &  &
$\ddot y= \displaystyle{\frac{\mu}{\eta}\dot x^2+\left (2\frac{\dot \mu}
{\eta}-\frac{\mu \dot \eta}{\eta^2}+\mu f(t)\right )\dot x} $\\
& & & $+\displaystyle{\left (\frac{\ddot \mu}{\eta}-\frac{\mu \ddot \eta}
{\eta^2}-2\frac{\dot \eta \dot \mu}{\eta^2}\right )\ln (\eta \dot x+\dot \eta)
+g(t)}$\\
